library(ggplot2)
library(dplyr)

plot <- ggplot(
  df,
  aes(
    x = list_name,
    y = io500_score,
    color = list_name,
    text = text
  )) + 
  geom_jitter(size = 1, width = 0.3) +
  scale_colour_manual(
    values = c(
      "#ec563c",
      "#1d2a30",
      "#ec563c",
      "#ec563c",
      "#1d2a30",
      "#ec563c",
      "#1d2a30",
      "#ec563c",
      "#1d2a30"
    )
  ) +
  xlab("") +
  ylab("IO500 Score") + 
  theme_bw() +
  theme(
    legend.position = "none",
    plot.title = element_text(size = 10),
    axis.title.y = element_text(vjust = 0.4), 
    strip.background = element_rect(colour = NA, fill = NA)
  )

plot
